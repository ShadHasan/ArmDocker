### Run docker mounting `firmware` and `libmali` and `mali device `

sudo docker run -it --rm \
    --name="temp" \
    --device /dev/mali0:/dev/mali0 \
    -v /lib/firmware/mali_csffw.bin:/lib/firmware/mali_csffw.bin:ro \
    -v /usr/lib/aarch64-linux-gnu/libmali.so:/usr/lib/libmali.so:ro \
    dev_kit /bin/bash
